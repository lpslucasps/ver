
# no scope
on_monthly_pulse = {
	events = {
		vermain.11 # Removes no longer applicable modifiers
		vermain.12 # Reapply modifiers
	}
}

# this = country
on_monthly_pulse_country = {
	events = {
		vermain.10 # Call ver.11, which transform Vassals into custom subject types
		vermain.13 # Look for machine empires with Parallel Processors and apply bonuses.
		vermain.2 # Select all countries which are taxing its subjects
		vermain.23 # Removes bonus trade value when not applicable anymore
		vermain.24 # Apply taxes to any newly-acquired subject.
	}
}

# A war has ended
# Root = Loser
# From = Main Winner
on_war_ended = {
	events = {
		ver.10 # Let player choose which type of subject the new vassal should become
	}
}

	}
